1) SELECT C.FIRST_NAME,C.LAST_NAME,P.PACK_ID,P.SPEED
FROM CUSTOMERS C JOIN PACKAGES P
ON (P.PACK_ID = C.PACK_ID);

2) SELECT C.FIRST_NAME,C.LAST_NAME,P.PACK_ID,P.SPEED
FROM CUSTOMERS C , PACKAGES P
WHERE P.PACK_ID = 22 OR P.PACK_ID = 27
ORDER BY C.LAST_NAME ASC;

3) SELECT P.PACK_ID,P.SPEED,P.MONTHLY_PAYMENT,S.SECTOR_NAME
FROM PACKAGES P , SECTORS S;

4) SELECT C.FIRST_NAME||C.LAST_NAME AS CUSTOMER_NAME,P.PACK_ID,P.SPEED,P.MONTHLY_PAYMENT,S.SECTOR_NAME
FROM CUSTOMERS C,PACKAGES P,SECTORS S;

5) SELECT C.FIRST_NAME||C.LAST_NAME AS CUSTOMER_NAME,P.PACK_ID,P.SPEED,P.MONTHLY_PAYMENT,S.SECTOR_NAME
FROM CUSTOMERS C,PACKAGES P,SECTORS S
WHERE S.SECTOR_NAME = 'Business';

6) SELECT C.LAST_NAME,C.FIRST_NAME,C.JOIN_DATE,P.PACK_ID,P.SPEED,S.SECTOR_NAME
FROM CUSTOMERS C,PACKAGES P,SECTORS S
WHERE S.SECTOR_NAME = 'Private'
AND TO_CHAR(C.JOIN_DATE,'YYYY')='2006'; 

7) SELECT P.PACK_ID,P.SPEED,P.MONTHLY_PAYMENT,G.GRADE_NAME
FROM PACKAGES P,PACK_GRADES G
WHERE P.PACK_ID = G.GRADE_ID;

8) SELECT C.FIRST_NAME,C.LAST_NAME,P.SPEED,P.MONTHLY_PAYMENT
FROM CUSTOMERS C,PACKAGES P;

9) SELECT C.FIRST_NAME,C.LAST_NAME,P.SPEED,P.MONTHLY_PAYMENT
FROM CUSTOMERS C , PACKAGES P
WHERE P.PACK_ID =C.PACK_ID(+) ;

10) SELECT C.FIRST_NAME,C.LAST_NAME,P.SPEED,P.MONTHLY_PAYMENT
FROM CUSTOMERS C , PACKAGES P
WHERE P.PACK_ID(+) =C.PACK_ID;

11) SELECT C.FIRST_NAME,C.LAST_NAME,P.SPEED,P.MONTHLY_PAYMENT
FROM CUSTOMERS C FULL OUTER JOIN  PACKAGES P
ON P.PACK_ID =C.PACK_ID;

12) SELECT C.LAST_NAME,C.FIRST_NAME,C.PACK_ID
FROM CUSTOMERS C,CUSTOMERS P
WHERE C.PACK_ID=P.PACK_ID AND P.FIRST_NAME='Amado' AND P.LAST_NAME='Taylor';

13) SELECT C.LAST_NAME,C.FIRST_NAME,C.MONTHLY_DISCOUNT
FROM CUSTOMERS C,CUSTOMERS P
WHERE C.MONTHLY_DISCOUNT<P.MONTHLY_DISCOUNT
AND P.CUSTOMER_ID = 103;

14) SELECT P1.PACK_ID,P1.SPEED
FROM PACKAGES P1 , PACKAGES P2
WHERE P1.SPEED = P2.SPEED AND P2.PACK_ID = 10;
